import java.util.Scanner;
abstract class Shape  {
	double a;
	double b;
	abstract void printArea();
}
class Rectangle extends Shape{
	double l; 
	double br;
  Rectangle(double a, double b){
	l=a;
	br=b;
}

	void printArea(){
		System.out.println("The Area of the rectangle is: "+l*br);
}
}
class Triangle extends Shape{
	double h; 
	double b;
	Triangle(double a, double b){
		h=a;
		this.b=b;
	}
	void printArea(){
		System.out.println("The Area of the triangle is: "+(h*b)/2.0);
	}
}
class Circle extends Shape{
	double r;
	Circle(double r){
		this.r=r;
	}
	void printArea(){
		System.out.println("The area of the Circle is: "+ 3.14159265*r*r);	
	
	}
}
class ShapeDemo{
	public static void main(String args[]){
		System.out.println("Name: Amith R");
		System.out.println("USN: 1BM23CS028");
		Rectangle r = new Rectangle(9,5);
		Triangle t = new Triangle(10,2);
		Circle c = new Circle(5);

		r.printArea();
		t.printArea();
		c.printArea();
		
	}
}
